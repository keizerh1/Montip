// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract Montip is ReentrancyGuard {
    struct Tip {
        address sender;
        address recipient;
        address token;
        uint256 amount;
        uint256 unlockTime;
        uint256 claimedAmount;
        uint256 dailyUnlockPercent;
        bool isProgressive;
    }
    
    mapping(uint256 => Tip) public tips;
    uint256 public tipCount;
    
    event TipSent(
        uint256 indexed tipId,
        address indexed sender,
        address indexed recipient,
        address token,
        uint256 amount,
        uint256 unlockTime,
        bool isProgressive,
        uint256 dailyUnlockPercent
    );
    
    event TipWithdrawn(
        uint256 indexed tipId,
        address indexed recipient,
        uint256 amount
    );
    
    function sendTip(
        address recipient,
        address token,
        uint256 amount,
        uint256 unlockDelay,
        bool isProgressive,
        uint256 dailyUnlockPercent
    ) external payable nonReentrant {
        require(recipient != address(0), "Invalid recipient");
        require(amount > 0, "Amount must be positive");
        
        uint256 unlockTime = block.timestamp + unlockDelay;
        
        if (token == address(0)) {
            require(msg.value == amount, "ETH amount mismatch");
        } else {
            IERC20(token).transferFrom(msg.sender, address(this), amount);
        }
        
        if (isProgressive) {
            require(dailyUnlockPercent > 0 && dailyUnlockPercent <= 100, "Invalid unlock percent");
        }
        
        tipCount++;
        tips[tipCount] = Tip({
            sender: msg.sender,
            recipient: recipient,
            token: token,
            amount: amount,
            unlockTime: unlockTime,
            claimedAmount: 0,
            dailyUnlockPercent: dailyUnlockPercent,
            isProgressive: isProgressive
        });
        
        emit TipSent(
            tipCount,
            msg.sender,
            recipient,
            token,
            amount,
            unlockTime,
            isProgressive,
            dailyUnlockPercent
        );
    }
    
    function withdrawTip(uint256 tipId) external nonReentrant {
        Tip storage tip = tips[tipId];
        require(tip.recipient == msg.sender, "Not the recipient");
        require(block.timestamp >= tip.unlockTime, "Tip is still locked");
        
        uint256 availableAmount;
        if (tip.isProgressive) {
            uint256 daysPassed = (block.timestamp - tip.unlockTime) / 1 days;
            uint256 maxClaimable = tip.amount * (tip.dailyUnlockPercent * daysPassed) / 100;
            availableAmount = maxClaimable - tip.claimedAmount;
        } else {
            availableAmount = tip.amount - tip.claimedAmount;
        }
        
        require(availableAmount > 0, "No funds available to withdraw");
        
        tip.claimedAmount += availableAmount;
        
        if (tip.token == address(0)) {
            payable(msg.sender).transfer(availableAmount);
        } else {
            IERC20(tip.token).transfer(msg.sender, availableAmount);
        }
        
        emit TipWithdrawn(tipId, msg.sender, availableAmount);
    }
    
    function getAvailableAmount(uint256 tipId) public view returns (uint256) {
        Tip storage tip = tips[tipId];
        if (block.timestamp < tip.unlockTime) return 0;
        
        if (tip.isProgressive) {
            uint256 daysPassed = (block.timestamp - tip.unlockTime) / 1 days;
            uint256 maxClaimable = tip.amount * (tip.dailyUnlockPercent * daysPassed) / 100;
            return maxClaimable - tip.claimedAmount;
        } else {
            return tip.amount - tip.claimedAmount;
        }
    }
}
